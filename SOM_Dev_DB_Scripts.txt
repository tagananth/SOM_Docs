CREATE TABLE customer (
	customerid	bigint,
	customername	text,
	active		boolean,
	comments	text,
	addressline1	text,
	addressline2	text,
	addressline3	text,
	city		text,
	state		text,
	country		text,
	zipcode		text,
	createdby	bigint,
	createdbyname	text,
	createddate	timestamp,
	modifiedby	bigint,
	modifiedbyname	text,
	modifieddate	timestamp,
	status		boolean,
	PRIMARY KEY (customerid)
);

	createddate	timestamp,

CREATE TABLE messages (
	messageid	bigint,
	sendid		text,
	instanceid	text,
	customerid	bigint,
	userid		bigint,
	username	text,
	groupid		bigint,
	groupname	text,
	configurationid	bigint,
	configurationname	text,
	type		text,	//Email/SMS

	emailid		text, 
	mobileno	text,
	templateid	text,

	modifiedbyname	text,
